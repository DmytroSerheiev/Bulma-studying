Syntax

{property}{sides}-{size} for all breakpoints
                    
        Properties:
        m - margin
        p - padding
        
        Sides:
        t - top
        r - right
        b - bottom
        l - left
        x - left and right
        y - top and bottom
        [blank] - all four sides
        
        Sizes:
        0 - 0
        1 - 0.25rem
        2 - 0.5rem
        3 - 0,75rem
        4 - 1rem
        5 - 1.5rem
        6 - 3rem
